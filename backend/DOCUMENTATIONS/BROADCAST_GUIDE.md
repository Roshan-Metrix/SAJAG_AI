# Admin Broadcast Route Testing Guide

This guide explains how to test the **admin broadcast WebSocket endpoint** that allows an admin to send real‑time notifications to all active citizens and rescue teams.

## Endpoint
- **URL:** `ws://<host>:8000/ws/admin-broadcast`
- **Query param:** `token` – JWT token of an admin user (must contain `role: "admin"`).

## Manual Testing with `websocat`
```bash
# Install websocat (if not installed)
cargo install websocat

# Replace <ADMIN_JWT> with a valid admin token.
websocat "ws://localhost:8000/ws/admin-broadcast?token=<ADMIN_JWT>"
```
After the connection is established, type a JSON payload and press **Enter**. Example payload:
```json
{ "type": "emergency_alert", "alert_type": "fire", "message": "Wildfire reported near sector 7", "location": { "lat": 27.7172, "lon": 85.3240 } }
```
All connected citizen and rescue‑team clients should receive the same JSON message.

async def websocket_client(uri, token, received: list):
    async with httpx.AsyncClient() as async_client:
        async with async_client.ws_connect(f"{uri}?token={token}") as ws:
            # Wait for a single message
            msg = await ws.receive_json()
            received.append(msg)

@pytest.mark.asyncio
async def test_admin_broadcast():
    # Start two mock clients: one citizen and one rescue team
    citizen_msgs = []
    rescue_msgs = []
    admin_token = get_admin_token()

    # Connect mock clients to their respective WS endpoints
    citizen_task = asyncio.create_task(
        websocket_client("ws://localhost:8000/ws/notifications", admin_token, citizen_msgs)
    )
    rescue_task = asyncio.create_task(
        websocket_client("ws://localhost:8000/ws/rescue-team-location", admin_token, rescue_msgs)
    )

    # Give the connections a moment to register
    await asyncio.sleep(0.1)

    # Connect admin broadcast WS and send a message
    async with httpx.AsyncClient() as ac:
        async with ac.ws_connect(f"ws://localhost:8000/ws/admin-broadcast?token={admin_token}") as admin_ws:
            broadcast = {
                "type": "emergency_alert",
                "alert_type": "flood",
                "message": "River overflow in zone 3",
                "location": {"lat": 27.7, "lon": 85.3},
            }
            await admin_ws.send_json(broadcast)

    # Wait for mock clients to receive the broadcast
    await asyncio.wait([citizen_task, rescue_task], timeout=2)

    # Verify both received the same payload
    assert citizen_msgs[0] == broadcast
    assert rescue_msgs[0] == broadcast
